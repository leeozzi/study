            @Override
            public int compare(String o1, String o2) {
                if(o1.length() != o2.length())

                    return o1.compareTo(o2);
            }

            String word = br.readLine();
            if(!pq.contains(word))
                pq.offer(word);
        }

        while(!pq.isEmpty()) {
            System.out.println(pq.poll());
        }
    }
